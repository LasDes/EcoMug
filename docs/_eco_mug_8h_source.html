33 class EMRandom {
34 public:
35  EMRandom() {
36  std::random_device rd;
37  std::mt19937 gen(rd());
38  std::uniform_int_distribution<uint64_t> dis(0, std::numeric_limits<uint64_t>::max());
39  s[0] = dis(gen);
40  s[1] = dis(gen);
41  };
42 
43  void SetSeed(uint64_t seed) {
44  s[0] = seed;
45  s[1] = seed;
46  };
47 
48  double GenerateRandomDouble() {
49  uint64_t x = next();
50  return to_double(x);
51  };
52 
53  double GenerateRandomDouble(double x1, double x2) {
54  return (x2-x1)*GenerateRandomDouble()+x1;
55  };
56 
57  int64_t rotl(const uint64_t x, int k) {
58  return (x << k) | (x >> (64 - k));
59  };
60 
61  uint64_t next() {
62  const uint64_t s0 = s[0];
63  uint64_t s1 = s[1];
64  const uint64_t result = s0 + s1;
65  s1 ^= s0;
66  s[0] = rotl(s0, 55) ^ s1 ^ (s1 << 14);
67  s[1] = rotl(s1, 36);
68  return result;
69  };
70 
71  double to_double(uint64_t x) {
72  union U {
73  uint64_t i;
74  double d;
75  };
76  U u = { UINT64_C(0x3FF) << 52 | x >> 12 };
77  return u.d - 1.0;
78  };
79 
80  uint64_t s[2];
81 };

88 class EMMaximization {
89 private:
90  EMRandom mRandom;
91  size_t mPopSize;
92  size_t mNIter;
93  int mGenMethod; // 0 = sky, 1 = cylinder, 2 = hspere
94  double m_a;
95  double m_a2;
96  std::vector<std::vector<double> > mRanges;
97  std::vector<std::vector<double> > mPopulation;
98  double mBestCost;
99  std::vector<double> mBestSolution;
100  std::function<double(double, double)> mFunc;
101 
102 public:
103  EMMaximization(const EMRandom& random, int genMethod) : mRandom(random), mPopSize(200),
104  mNIter(500), mGenMethod(genMethod), m_a(0.), m_a2(0.), mBestCost(-1.) {
105  mFunc = &DefaultJ;
106  };
108 
109  static double DefaultJ(double p, double theta) {
110  double n = std::max(0.1, 2.856-0.655*log(p));
111  return 1600*pow(p, 0.279)*pow(cos(theta), n);
112  }
114 
115  void SetParameters(double minP, double maxP, double minTheta, double maxTheta) {
116  mRanges.push_back({minP, maxP});
117  mRanges.push_back({minTheta, maxTheta});
118  }
120 
121  void SetParameters(double minP, double maxP, double minTheta, double maxTheta, double minPhi, double maxPhi) {
122  mRanges.push_back({minP, maxP});
123  mRanges.push_back({minTheta, maxTheta});
124  mRanges.push_back({minPhi, maxPhi});
125  mRanges.push_back({0, M_PI/2.});
126  }
128 
129  void SetFunction(std::function<double(double, double)> func) {
130  mFunc = func;
131  }
133 
134  double SkyFunc(double p, double theta) {
135  return mFunc(p, theta)*cos(theta)*sin(theta);
136  }
138 
139  double CylFunc(double p, double theta) {
140  return mFunc(p, theta)*pow(sin(theta), 2);
141  }
143 
144  double HSFunc(double p, double theta, double phi, double theta0) {
145  return mFunc(p, theta)*(sin(theta0)*sin(theta)*cos(phi) + cos(theta0)*cos(theta))*sin(theta);
146  }
148 
149  double Evaluate(std::vector<double> &v) {
150  if (mGenMethod == 0) {
151  return SkyFunc(v[0], v[1]);
152  } else if (mGenMethod == 1) {
153  return CylFunc(v[0], v[1]);
154  } else {
155  return HSFunc(v[0], v[1], v[2], v[3]);
156  }
157  return -1;
158  }
160 
161  void Evaluate() {
162  double value;
163  for (size_t i = 0; i < mPopSize; ++i) {
164  value = Evaluate(mPopulation[i]);
165  if (value > mBestCost) {
166  mBestCost = value;
167  mBestSolution = mPopulation[i];
168  }
169  }
170  }
172 
173  void Init() {
174  size_t dim = mRanges.size();
175  mPopulation.resize(mPopSize);
176  for (size_t i = 0; i < mPopSize; ++i) {
177  mPopulation[i].resize(dim);
178  for (size_t j = 0; j < dim; ++j) {
179  mPopulation[i][j] = mRandom.GenerateRandomDouble(mRanges[j][0], mRanges[j][1]);
180  }
181  }
182  }
184 
185  void UpdateParameters(size_t t) {
186  m_a = 2. - t*(2./mNIter);
187  m_a2 = -1. + t*((-1.)/mNIter);
188  }
190 
191  void Move() {
192  double r1, r2, A, C, b, l, rw, p, D_tmp, D_best, distance;
193  std::vector<double> tmp;
194  for (size_t i = 0; i < mPopulation.size(); ++i) {
195  r1 = mRandom.GenerateRandomDouble();
196  r2 = mRandom.GenerateRandomDouble();
197  A = 2*m_a*r1-m_a;
198  C = 2*r2;
199  b = 1.;
200  l = (m_a2-1)*mRandom.GenerateRandomDouble()+1;
201  p = mRandom.GenerateRandomDouble();
202 
203  for (size_t j = 0; j < mPopulation[0].size(); ++j) {
204  if (p < 0.5) {
205  if (fabs(A) >= 1) {
206  rw = floor(mRandom.GenerateRandomDouble()*mPopulation.size());
207  tmp = mPopulation[rw];
208  D_tmp = fabs(C*tmp[j] - mPopulation[i][j]);
209  mPopulation[i][j] = tmp[j] - A*D_tmp;
210  } else {
211  D_best = fabs(C*mBestSolution[j] - mPopulation[i][j]);
212  mPopulation[i][j] = mBestSolution[j]-A*D_best;
213  }
214  } else {
215  distance = fabs(mBestSolution[j] - mPopulation[i][j]);
216  mPopulation[i][j] = distance*exp(b*l)*cos(l*2*M_PI) + mBestSolution[j];
217  }
218  if (mPopulation[i][j] < mRanges[j][0]) mPopulation[i][j] = mRanges[j][0];
219  if (mPopulation[i][j] > mRanges[j][1]) mPopulation[i][j] = mRanges[j][1];
220  }
221  }
222  }
224 
225  double Maximize() {
226  Init();
227  Evaluate();
228  for (size_t iter = 1; iter < mNIter; ++iter) {
229  UpdateParameters(iter);
230  Move();
231  Evaluate();
232  }
233  return mBestCost;
234  }
235 };

556 public:
560  void Generate() {
561  mAccepted = false;
562 
563  if (mMaxJ[mGenMethod] < 0) ComputeMaximum();
564 
565  // Sky or cylinder generation
566  if (mGenMethod == Sky || mGenMethod == Cylinder) {
567  // Generation of the momentum and theta angle
568  while (!mAccepted) {
569  mRandAccRej = mRandom.GenerateRandomDouble();
570  mGenerationTheta = mRandom.GenerateRandomDouble(mMinimumTheta, mMaximumTheta);
571  mGenerationMomentum = GenerateMomentumF1();
572  mN = 2.856-0.655*log(mGenerationMomentum);
573  if (mN < 0.1) mN = 0.1;
574 
575  if (mGenMethod == Sky) {
576  mJPrime = 1600*pow(mGenerationMomentum, 0.279)*pow(cos(mGenerationTheta), mN+1)*sin(mGenerationTheta);
577  if (mMaxJ[mGenMethod]*mRandAccRej < mJPrime) mAccepted = true;
578  }
579 
580  if(mGenMethod == Cylinder) {
581  mJPrime = 1600*pow(mGenerationMomentum, 0.279)*pow(cos(mGenerationTheta), mN)*pow(sin(mGenerationTheta), 2);
582  if (mMaxJ[mGenMethod]*mRandAccRej < mJPrime) mAccepted = true;
583  }
584  }
585  mGenerationTheta = M_PI - mGenerationTheta;
586 
587  // Generation of the position and phi angle
588  if (mGenMethod == Sky) {
589  GeneratePositionSky();
590  mGenerationPhi = mRandom.GenerateRandomDouble(mMinimumPhi, mMaximumPhi);
591  }
592  if (mGenMethod == Cylinder) {
593  mAccepted = false;
594  GeneratePositionCylinder();
595  while (!mAccepted) {
596  mRandAccRej = mRandom.GenerateRandomDouble();
597  mGenerationPhi = mRandom.GenerateRandomDouble(mMinimumPhi, mMaximumPhi);
598  if (mRandAccRej < fabs(cos(mGenerationPhi))) mAccepted = true;
599  }
600  mGenerationPhi = mGenerationPhi + mPhi0;
601  if (mGenerationPhi >= 2.*M_PI) mGenerationPhi -= 2.*M_PI;
602 
603  // Check if the muon is inward
604  if (sin(mGenerationTheta)*cos(mGenerationPhi)*mGenerationPosition[0] + sin(mGenerationTheta)*sin(mGenerationPhi)*mGenerationPosition[1] > 0) Generate();
605  }
606  }
607 
608  // Half-sphere generation
609  if (mGenMethod == HSphere) {
610  // Generation point on the half-sphere
611  mPhi0 = 2.*M_PI*mRandom.GenerateRandomDouble();
612  while (!mAccepted) {
613  mRandAccRej = mRandom.GenerateRandomDouble();
614  mTheta0 = acos(mRandom.GenerateRandomDouble());
615  mGenerationTheta = mRandom.GenerateRandomDouble(mMinimumTheta, mMaximumTheta);
616  mGenerationPhi = mRandom.GenerateRandomDouble(mMinimumPhi, mMaximumPhi);
617  mGenerationMomentum = GenerateMomentumF1();
618  mN = 2.856-0.655*log(mGenerationMomentum);
619  if (mN < 0.1) mN = 0.1;
620 
621  mJPrime = 1600*pow(mGenerationMomentum, 0.279)*pow(cos(mGenerationTheta), mN)*(sin(mGenerationTheta)*sin(mTheta0)*cos(mGenerationPhi)+cos(mGenerationTheta)*cos(mTheta0))*sin(mGenerationTheta);
622  if (mMaxJ[mGenMethod]*mRandAccRej < mJPrime) mAccepted = true;
623  }
624 
625  mGenerationPosition[0] = mHSphereRadius*sin(mTheta0)*cos(mPhi0) + mHSphereCenterPosition[0];
626  mGenerationPosition[1] = mHSphereRadius*sin(mTheta0)*sin(mPhi0) + mHSphereCenterPosition[1];
627  mGenerationPosition[2] = mHSphereRadius*cos(mTheta0) + mHSphereCenterPosition[2];
628 
629  mGenerationTheta = M_PI - mGenerationTheta;
630  mGenerationPhi = mGenerationPhi + mPhi0;
631  if (mGenerationPhi >= 2*M_PI) mGenerationPhi -= 2*M_PI;
632 
633  mGenerationPhi += M_PI;
634  if (mGenerationPhi >= 2*M_PI) mGenerationPhi -= 2*M_PI;
635  }
636 
637  // Generate the charge
638  mCharge = (mDiscDistC(mEngineC) == 0) ? 1 : -1;
639  };

648 
649  if (mMaxCustomJ[mGenMethod] < 0) ComputeMaximumCustomJ();
650 
651  // Sky or cylinder generation
652  if (mGenMethod == Sky || mGenMethod == Cylinder) {

666  if (mMaxCustomJ[mGenMethod]*mRandAccRej < mJPrime) mAccepted = true;
667  }
668  }
669  mGenerationTheta = M_PI - mGenerationTheta;
670 
671  // Generation of the position and phi angle
672  if (mGenMethod == Sky) {
673  GeneratePositionSky();
674  mGenerationPhi = mRandom.GenerateRandomDouble(mMinimumPhi, mMaximumPhi);
675  }
676  if (mGenMethod == Cylinder) {
677  mAccepted = false;
678  GeneratePositionCylinder();
679  while (!mAccepted) {
680  mRandAccRej = mRandom.GenerateRandomDouble();
681  mGenerationPhi = mRandom.GenerateRandomDouble(mMinimumPhi, mMaximumPhi);
682  if (mRandAccRej < fabs(cos(mGenerationPhi))) mAccepted = true;
683  }
684  mGenerationPhi = mGenerationPhi + mPhi0;
685  if (mGenerationPhi >= 2.*M_PI) mGenerationPhi -= 2.*M_PI;
686 
687  // Check if the muon is inward
688  if (sin(mGenerationTheta)*cos(mGenerationPhi)*mGenerationPosition[0] + sin(mGenerationTheta)*sin(mGenerationPhi)*mGenerationPosition[1] > 0) Generate();
689  }
690  }
691 
692  // Half-sphere generation
693  if (mGenMethod == HSphere) {
694  // Generation point on the half-sphere
695  mPhi0 = 2.*M_PI*mRandom.GenerateRandomDouble();
696  while (!mAccepted) {
697  mRandAccRej = mRandom.GenerateRandomDouble();
698  mTheta0 = acos(mRandom.GenerateRandomDouble());
699  mGenerationTheta = mRandom.GenerateRandomDouble(mMinimumTheta, mMaximumTheta);
700  mGenerationPhi = mRandom.GenerateRandomDouble(mMinimumPhi, mMaximumPhi);
